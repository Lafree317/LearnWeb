<script type="text/javascript">
    window.onload = function () {
            var oP = document.getElementsByTagName("p")[0];
            document.onkeydown = function (event) {
                var event = event || window.event;
                oP.innerHTML = event.keyCode;
                return false
            }
        }
    </script>

这里将显示您按下的 KeyCode
